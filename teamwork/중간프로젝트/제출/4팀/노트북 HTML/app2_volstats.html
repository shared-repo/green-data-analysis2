    acf_change = sm.tsa.acf(vol_est[i]['change'].dropna(), nlags=100)[1:]
    acf_hv = sm.tsa.acf(vol_est[i]['hv'].dropna(), nlags=100)[1:]
    acf_ewma = sm.tsa.acf(vol_est[i]['ewma'].dropna(), nlags=100)[1:]
    if vol_est[i]['iv'].dropna().empty == False:
        acf_iv = sm.tsa.acf(vol_est[i]['iv'].dropna(), nlags=100)[1:]
    else:
        acf_iv = sm.tsa.acf(vol_est[i]['iv'], nlags=100)[1:]

    # ACF 1
    plt.figure(figsize=(13, 3))
    plt.subplot(1, 4, 1)
    plt.stem(range(1, len(acf_change) + 1), acf_change, markerfmt='o', basefmt='gray')
    plt.axhline(0.1, color='black')
    plt.axhline(-0.1, color='black')
    plt.title(f'Change of {bench_code_lst[i]}')
    plt.xlabel('Lag')
    plt.ylabel('ACF')

    # ACF 2
