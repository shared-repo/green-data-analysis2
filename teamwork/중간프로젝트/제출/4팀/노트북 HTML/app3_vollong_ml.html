def preprocessing_net(df):
    # datetime을 제외한 모든 컬럼을 float으로 변경
    for col in df.columns:
        if col != 'datetime':
            try:
                df[col] = df[col].str.replace(',', '').astype(float)
            except:
                df[col] = df[col].astype(float)
    return df

def df_abs(df):
    # datetime을 제외한 모든 열에 절대값 적용
    for col in df.columns:
        if col != 'datetime':
            df[col] = df[col].abs()
    return df
