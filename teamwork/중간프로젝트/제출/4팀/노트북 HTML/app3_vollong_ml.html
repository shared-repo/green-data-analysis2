def preprocessing_net(df):
    # datetime    float 
    for col in df.columns:
        if col != 'datetime':
            try:
                df[col] = df[col].str.replace(',', '').astype(float)
            except:
                df[col] = df[col].astype(float)
    return df

def df_abs(df):
    # datetime     
    for col in df.columns:
        if col != 'datetime':
            df[col] = df[col].abs()
    return df
