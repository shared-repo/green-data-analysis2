# Function to categorize the change
def categorize_change(x):
    if x < 0:
        if x >= mean - std:
            return 'Negative Small Change'
        elif x >= q25:
            return 'Negative Medium Change'
        else:
            return 'Negative Large Change'
    else:
        if x <= mean + std:
            return 'Positive Small Change'
        elif x <= q75:
            return 'Positive Medium Change'
        else:
            return 'Positive Large Change'

from sklearn.svm import SVC
svc_scores = []
best_scores = {}
for column in ['change_category_dowjones', 'change_category_nasdaq', 'change_category_s&p500']:
    best_score = 0
    for gamma in np.arange(-10,10):
        for C in np.arange(-10,10):
            #     SVC 
            svm = SVC(gamma=10**int(gamma), C=10**int(C))
            svm.fit(X_train, y_train[column])
            score = svm.score(X_test, y_test[column]) #   
            svc_scores.append((10**int(gamma), 10**int(C), column, score ))
            if score > best_score: #       
                best_score = score
                best_parameters = {'C': 10**int(C), 'gamma': 10**int(gamma)}
    best_scores[column] = {
        'Scores': best_score,
        'BP' : best_parameters
        }
